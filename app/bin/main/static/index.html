ភាសាកម្មវិធីខ្មែរ - Khmer Programming Language

📝 កូដ (Code)

📋 លទ្ធផល (Output)

📚 មេរៀន & ឧទាហរណ៍

🔢 ប្រភេទទិន្ន័យ (Data Types)

// ចំនួន (Integer)
តាង ក = 10;

// ចំនួន ទស (Double)  
តាង ខ = 3.14;

// តួអក្សរ (String)
តាង គ = "សួស្តី";

🔄 ប្រមាណវិធី (Operators)

// បូក (+) ដក (-) គុណ (×) ចែក (÷)
តាង ក = 10 + 5;   // = 15
តាង ខ = 10 - 5;   // = 5
តាង គ = 10 * 2;   // = 20
តាង ឃ = 10 / 2;   // = 5

🔀 លក្ខណៈ (Conditions)

// បើសិន (if) ម្យ៉ាងវិញ (else if) ផ្សេង (else)
បើសិន ក > 10 {
    ហៅ "ក ធំ";
}
ម្យ៉ាងវិញ បើសិន ក == 10 {
    ហៅ "ក ស្មើ";
}
ផ្សេង {
    ហៅ "ក តូច";
}

📊 ប្រៀបធៀប (Comparison)

// == ស្មើ  != មិនស្មើ
// >  ធំជាង  < តូចជាង  
// >= ធំជាង ឬ ស្មើ  <= តូចជាង ឬ ស្មើ

បើសិន ក >= 18 {
    ហៅ "អាយុគ្រប់";
}
